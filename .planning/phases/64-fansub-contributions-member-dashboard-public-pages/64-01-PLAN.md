    2. historical_leader: SELECT fgmr.id FROM fansub_group_member_roles fgmr JOIN fansub_group_members fgm ON fgm.id = fgmr.fansub_group_member_id WHERE fgm.member_id = $1 AND fgmr.role_code IN ('leader', 'founder') LIMIT 1 — wenn Ergebnis: UpsertMemberBadge mit badge_code "historical_leader"

    3. long_term_member: SELECT 1 FROM fansub_group_members WHERE member_id = $1 AND (left_year - joined_year >= 5 OR (joined_year IS NOT NULL AND left_year IS NULL)) LIMIT 1 — wenn Ergebnis: UpsertMemberBadge mit badge_code "long_term_member"
