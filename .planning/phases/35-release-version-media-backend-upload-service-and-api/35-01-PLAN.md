```dockerfile
FROM golang:1.25-alpine AS builder

# CGO build tools required for govips (libvips wrapper)
RUN apk add --no-cache build-base pkgconfig vips-dev

WORKDIR /app

COPY go.mod go.sum ./
RUN go mod download

COPY . .
# CGO_ENABLED=1 required for govips — do NOT change back to 0
RUN CGO_ENABLED=1 GOOS=linux go build -o server ./cmd/server
RUN CGO_ENABLED=1 GOOS=linux go build -o migrate ./cmd/migrate

FROM alpine:3.20
WORKDIR /app

# Runtime shared library for govips dynamic linking
RUN apk add --no-cache vips

COPY --from=builder /app/server ./server
COPY --from=builder /app/migrate ./migrate

EXPOSE 8092
CMD ["./server"]
```

- Runtime stage: `RUN apk add --no-cache vips` added before COPY

Step 2 — Add govips Startup/Shutdown to main.go:

In `backend/cmd/server/main.go`, add the import:
```go
"github.com/davidbyttow/govips/v2/vips"
```

Insert these lines AFTER the ffmpeg availability check block and BEFORE `router := gin.New()`:
```go
// govips (libvips wrapper) must be initialized before any image processing calls.
// vips.Shutdown() is deferred here so it runs after server shutdown completes.
if err := vips.Startup(nil); err != nil {
    log.Fatalf("govips startup failed: %v", err)
}
defer vips.Shutdown()
```

The exact insertion point in main.go is after line ~47 (after `log.Printf("ffmpeg available at %s"...)`), before `router := gin.New()`.

Do NOT move or change any existing defer statements. The `defer vips.Shutdown()` should be placed after other defer statements so it runs first (LIFO order — vips shuts down before DB pool closes).
