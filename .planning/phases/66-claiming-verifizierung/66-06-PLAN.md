    - claimQueueLoading: boolean
    - claimQueueError: string | null

    Beim Laden der Seite (useEffect mit groupId): listPendingMemberClaims(groupId) → setPendingClaims

    Handler handleGenerateInvitation(memberId: number):
    - generateClaimInvitation(groupId, memberId)
    - setGeneratedInvites(prev => ({...prev, [memberId]: result.data}))

    Handler handleCopyLink(memberId: number, link: string):
    - navigator.clipboard.writeText(link)
    - setCopyStates(prev => ({...prev, [memberId]: true}))
    - setTimeout(() => setCopyStates(prev => ({...prev, [memberId]: false})), 1500)

    Handler handleVerifyClaim(claimId: number):
    - verifyMemberClaim(groupId, claimId)
    - setPendingClaims(prev => prev.filter(c => c.id !== claimId))
    - catch: setErrorMessage(readErrorMessage(err, "Claim konnte nicht bestätigt werden."))

    Handler handleRejectClaim(claimId: number, memberNick: string):
    - if (!window.confirm(`Claim für "${memberNick}" ablehnen?`)) return
    - rejectMemberClaim(groupId, claimId)
    - setPendingClaims(prev => prev.filter(c => c.id !== claimId))
