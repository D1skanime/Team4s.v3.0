  Repository-Struct: AnimeContributionsRepository mit db *pgxpool.Pool
  Constructor: NewAnimeContributionsRepository(db *pgxpool.Pool) *AnimeContributionsRepository

  Methoden:

  ListByFansubAndAnime(ctx, fansubGroupID int64, animeID int64) ([]AnimeContributionRow, error)
  - SELECT ac.*, ARRAY_AGG(acr.role_code) FILTER (WHERE acr.role_code IS NOT NULL) AS role_codes
    FROM anime_contributions ac
    LEFT JOIN anime_contribution_roles acr ON acr.anime_contribution_id = ac.id
    WHERE ac.fansub_group_id = $1 AND ac.anime_id = $2
    GROUP BY ac.id ORDER BY ac.created_at

  GetByID(ctx, id int64) (*AnimeContributionRow, error)
  - SELECT + LEFT JOIN anime_contribution_roles GROUP BY ac.id WHERE ac.id = $1
  - ErrNotFound bei pgx.ErrNoRows

  Create(ctx, fansubGroupID int64, animeID int64, input AnimeContributionInput) (*AnimeContributionRow, error)
  - Transaktion: 1) INSERT INTO anime_contributions RETURNING id, 2) INSERT INTO anime_contribution_roles fuer jeden RoleCodes-Eintrag (einzelne Inserts in Schleife oder multi-value INSERT), 3) GetByID im selben Tx-Kontext
  - Bei leerem RoleCodes-Slice: nur anime_contributions INSERT, keine Rollen-Inserts

  Update(ctx, id int64, input AnimeContributionPatchInput) (*AnimeContributionRow, error)
  - Wenn input.RoleCodes != nil: Transaktion — DELETE FROM anime_contribution_roles WHERE anime_contribution_id = $1, dann neue Rollen einfuegen
  - Andere Felder: UPDATE SET nur gesetzte Felder + updated_by = input.UpdatedBy, updated_at = NOW()
  - ErrNotFound wenn keine Zeile betroffen

  Delete(ctx, id int64) error
  - DELETE FROM anime_contributions WHERE id = $1 (anime_contribution_roles wird automatisch via CASCADE geloescht)
  - ErrNotFound wenn RowsAffected == 0

  ListPublicByAnime(ctx, animeID int64) ([]PublicContributionRow, error)
  - SELECT m.display_name, m.slug, ARRAY_AGG(acr.role_code) FILTER (...) AS role_codes, ac.started_year, ac.ended_year, (ac.status = 'confirmed') AS is_verified
    FROM anime_contributions ac
    JOIN hist_fansub_group_members hfgm ON hfgm.id = ac.fansub_group_member_id
    JOIN members m ON m.id = hfgm.member_id
    LEFT JOIN anime_contribution_roles acr ON acr.anime_contribution_id = ac.id
    WHERE ac.anime_id = $1 AND ac.is_public_on_anime_page = true AND hfgm.visibility = 'public'
    GROUP BY m.display_name, m.slug, ac.started_year, ac.ended_year, ac.status
    ORDER BY COALESCE(ac.started_year, 9999), m.display_name
    LIMIT 50

  ListPublicByFansub(ctx, fansubGroupID int64) ([]PublicContributionRow, error)
  - Analog zu ListPublicByAnime, aber WHERE ac.fansub_group_id = $1 AND ac.is_public_on_anime_page = true AND hfgm.visibility = 'public'
  - Keine Sichtbarkeits-Anforderung auf is_public_on_member_profile fuer Fansub-Seite

  ListPublicByMemberSlug(ctx, memberSlug string) ([]PublicContributionRow, error)
  - JOIN auf members WHERE members.slug = $1
  - WHERE ac.is_public_on_member_profile = true
  - LIMIT 50
