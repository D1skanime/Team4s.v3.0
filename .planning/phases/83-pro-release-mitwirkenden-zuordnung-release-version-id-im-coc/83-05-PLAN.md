```tsx
'use client'

import styles from './FansubEdit.module.css'

interface ContributorAvatarProps {
  name: string
  avatarUrl?: string | null
}

export function ContributorAvatar({ name, avatarUrl }: ContributorAvatarProps) {
  const initials = name
    .split(' ')
    .slice(0, 2)
    .map((s) => s[0]?.toUpperCase() ?? '')
    .join('')

  if (avatarUrl) {
    return <img src={avatarUrl} alt={name} className={styles.contributorAvatar} />
  }

  return (
    <span className={styles.contributorAvatarInitials} aria-label={name}>
      {initials}
    </span>
  )
}
```

**ReleaseContributionDrawer.tsx — Props-Interface:**
```typescript
interface ReleaseContributionDrawerProps {
  open: boolean
  fansubId: number
  animeId: number
  releaseVersionId: number
  releaseTitle: string
  onClose: () => void
  onSaved: () => void
}
```

**State:**
- `stagedRows: EffectiveContributionRow[]` — geladene + lokal geänderte Rollen
- `removedIds: number[]` — contribution_ids zum Löschen beim Speichern
- `addedRows: Array<{memberId: number, roleCodes: string[]}>` — neue Zuteilungen
- `loading: boolean`, `saving: boolean`, `error: string | null`
- `members: UnifiedGroupMember[]` — aus `listUnifiedGroupMembers`
- `isOverride: boolean` — aus meta.is_override
- Für "Hinzufügen": `addingRow: boolean`, `newMemberId: number | null`, `newRoleCode: string | null`

**Beim Öffnen (useEffect auf `open`):** Paralleles Laden via `Promise.all([listUnifiedGroupMembers(fansubId), listEffectiveContributionsForVersion(releaseVersionId, fansubId)])`. Fehlerbehandlung mit `setError`.

**Rollen-Liste (UI-SPEC Interaction Contract):**
- `Drawer` mit `title="Mitwirkende"`, `description=\`Rollen für ${releaseTitle}\``
- Kein separater "Vom Projekt geerbtes Team"-Block (D-07, C-NO-DIFF)
- Pro `stagedRows`-Eintrag: `div.contributionRow` mit `[roleLabel] · [ContributorAvatar] [memberName] · [⋯ Rolle ändern] [Trash2 Entfernen]`
- `Button variant="ghost" iconOnly size="sm" aria-label="Rolle ändern"` + `MoreHorizontal`-Icon (aus lucide-react)
- `Button variant="danger" iconOnly size="sm" aria-label="Mitwirkende entfernen"` + `Trash2`-Icon
- "Entfernen" entfernt Zeile aus `stagedRows` und fügt contribution_id zu `removedIds` hinzu (nur staged — kein API-Call)
- `Button variant="secondary"` + `Plus`-Icon: `"Rolle / Person hinzufügen"` → setzt `addingRow=true`
- Wenn `addingRow=true`: Inline-Zeile mit Person-Picker (`Select`+`FormField label="Person"`) + Rollen-Picker (`Select`+`FormField label="Rolle"`)
- `EmptyState` wenn `stagedRows.length === 0 && !addingRow`

**Footer (staged — C-STAGED):**
- `Button variant="ghost" onClick={handleClose} disabled={saving}` → "Abbrechen" (mit unsaved-changes-check: Modal-Confirm wenn staged-Änderungen existieren)
- `Button variant="primary" onClick={handleSave} loading={saving}` → "Speichern"

**handleSave-Algorithmus (Pitfall 4 beachten):**
1. `await Promise.all(removedIds.map(id => deleteAnimeContribution(fansubId, animeId, id)))` — löscht GENAU die versions-spezifischen IDs
2. `await Promise.all(addedRows.map(r => upsertAnimeContribution(fansubId, animeId, {member_id: r.memberId, role_codes: r.roleCodes, release_version_id: releaseVersionId})))` 
3. Bei Fehler: `setError("Speichern fehlgeschlagen. Bitte Eingaben prüfen und erneut versuchen.")`
4. Erfolg: `onSaved()`, `onClose()`

**Person-Picker:** Optionen aus `members` (UnifiedGroupMember — D-12: nur aktuelle Gruppe). Bereits in `stagedRows` enthaltene member_ids ausfiltern.

**Rollen-Picker:** Rollen-Liste aus `role_definitions`-Catalog. Statische Liste aus roleLabels (anime_contribution_roles context) OR dynamischer Endpoint. Empfehlung: Statische Liste der bekannten Rollen-Codes mit deutschen Labels (aus Migration 0085), da kein neuer Endpoint nötig.
