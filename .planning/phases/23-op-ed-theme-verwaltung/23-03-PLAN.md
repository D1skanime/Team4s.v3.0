// SaveVideoUpload validiert und speichert einen Video-Upload fĂĽr Theme-Assets.
// Erlaubte MIME-Typen: video/mp4, video/webm, video/x-matroska, video/x-msvideo, video/quicktime.
// Maximale DateigrĂ¶Ăźe: 500 MB.
func (s *MediaService) SaveVideoUpload(originalName string, data []byte) (*MediaSaveResult, error) {
    if len(data) == 0 {
        return nil, &MediaValidationError{Message: "datei ist leer"}
    }
    const maxVideoSize = 500 * 1024 * 1024 // 500 MB
    if int64(len(data)) > maxVideoSize {
        return nil, &MediaValidationError{Message: "video ist zu gross (max 500MB)"}
    }
    detectedMime := detectMimeType(data)
    allowed := map[string]string{
        "video/mp4":        "mp4",
        "video/webm":       "webm",
        "video/x-matroska": "mkv",
        "video/x-msvideo":  "avi",
        "video/quicktime":  "mov",
    }
    ext, ok := allowed[detectedMime]
    if !ok {
        return nil, &MediaValidationError{Message: fmt.Sprintf("nicht erlaubter video-typ: %s", detectedMime)}
    }
    filename := buildFilename(models.MediaKindThemeVideo, ext)
    absolutePath := filepath.Join(s.storageDir, filename)
    if err := os.MkdirAll(filepath.Dir(absolutePath), 0o755); err != nil {
        return nil, fmt.Errorf("create video directory: %w", err)
    }
    if err := os.WriteFile(absolutePath, data, fs.FileMode(0o644)); err != nil {
        return nil, fmt.Errorf("write video file: %w", err)
    }
    publicURL := fmt.Sprintf("%s/api/v1/media/files/%s", s.publicBaseURL, url.PathEscape(filename))
    return &MediaSaveResult{
        CreateInput: models.MediaAssetCreateInput{
            Filename:    filename,
            StoragePath: absolutePath,
            PublicURL:   publicURL,
            MimeType:    detectedMime,
            SizeBytes:   int64(len(data)),
        },
    }, nil
}

// ListReleaseThemeAssets gibt alle Theme-Video-VerknĂĽpfungen fĂĽr ein Release zurĂĽck.
func (r *AdminContentRepository) ListReleaseThemeAssets(ctx context.Context, releaseID int64) ([]models.AdminReleaseThemeAsset, error) {
    if releaseID <= 0 {
        return nil, ErrNotFound
    }
    // Verify release exists
    var exists bool
    if err := r.db.QueryRow(ctx, `SELECT EXISTS(SELECT 1 FROM fansub_releases WHERE id = $1)`, releaseID).Scan(&exists); err != nil {
        return nil, err
    }
    if !exists {
        return nil, ErrNotFound
    }
    rows, err := r.db.Query(ctx, `
        SELECT
            rta.release_id,
            rta.theme_id,
            tt.name AS theme_type_name,
            t.title AS theme_title,
            rta.media_id,
            ma.public_url,
            ma.mime_type,
            ma.size_bytes,
            rta.created_at
        FROM release_theme_assets rta
        JOIN themes t ON t.id = rta.theme_id
        JOIN theme_types tt ON tt.id = t.theme_type_id
        JOIN media_assets ma ON ma.id = rta.media_id
        WHERE rta.release_id = $1
        ORDER BY rta.created_at ASC
    `, releaseID)
    if err != nil {
        return nil, err
    }
    defer rows.Close()
    var result []models.AdminReleaseThemeAsset
    for rows.Next() {
        var a models.AdminReleaseThemeAsset
        if err := rows.Scan(&a.ReleaseID, &a.ThemeID, &a.ThemeTypeName, &a.ThemeTitle, &a.MediaID, &a.PublicURL, &a.MimeType, &a.SizeBytes, &a.CreatedAt); err != nil {
            return nil, err
        }
        result = append(result, a)
    }
    return result, rows.Err()
}

// CreateReleaseThemeAsset speichert zuerst den Media-Asset-Eintrag und verknĂĽpft ihn dann mit Release + Theme.
// Gibt ErrConflict zurĂĽck wenn die Kombination (release_id, theme_id) bereits existiert.
func (r *AdminContentRepository) CreateReleaseThemeAsset(ctx context.Context, input models.AdminReleaseThemeAssetCreateInput) (*models.AdminReleaseThemeAsset, error) {
    // Insert the release_theme_assets link
    _, err := r.db.Exec(ctx, `
        INSERT INTO release_theme_assets (release_id, theme_id, media_id)
        VALUES ($1, $2, $3)
    `, input.ReleaseID, input.ThemeID, input.MediaID)
    if err != nil {
        var pgErr *pgconn.PgError
        if errors.As(err, &pgErr) {
            switch pgErr.Code {
            case "23505": // unique_violation (PK conflict)
                return nil, ErrConflict
            case "23503": // foreign_key_violation (invalid release/theme/media)
                return nil, ErrConflict
            }
        }
        return nil, err
    }
    // Fetch the newly created record with joined names
    results, err := r.ListReleaseThemeAssets(ctx, input.ReleaseID)
    if err != nil {
        return nil, err
    }
    for _, a := range results {
        if a.ThemeID == input.ThemeID && a.MediaID == input.MediaID {
            return &a, nil
        }
    }
    return nil, ErrNotFound
}

// DeleteReleaseThemeAsset entfernt die VerknĂĽpfung zwischen Release und Theme.
// Das zugehĂ¶rige Media-Asset in media_assets wird NICHT gelĂ¶scht (kĂ¶nnte referenziert sein).
func (r *AdminContentRepository) DeleteReleaseThemeAsset(ctx context.Context, releaseID int64, themeID int64) error {
    if releaseID <= 0 || themeID <= 0 {
        return ErrNotFound
    }
    tag, err := r.db.Exec(ctx, `
        DELETE FROM release_theme_assets WHERE release_id = $1 AND theme_id = $2
    `, releaseID, themeID)
    if err != nil {
        return err
    }
    if tag.RowsAffected() == 0 {
        return ErrNotFound
    }
    return nil
}
