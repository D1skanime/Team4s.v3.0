**`useAdminAnimeThemes.test.ts`:**

Use `vi.mock('@/lib/api')` to mock all API helpers. Test that on mount the hook calls both `getAdminAnimeThemes` and `getAdminThemeTypes`. Test that `createTheme` calls `createAdminAnimeTheme` with the right arguments.

```typescript
import { describe, it, expect, vi, beforeEach } from 'vitest'
import { renderHook, act, waitFor } from '@testing-library/react'
import { useAdminAnimeThemes } from './useAdminAnimeThemes'
import * as api from '@/lib/api'

vi.mock('@/lib/api')

const mockThemes = { data: [] }
const mockTypes = { data: [{ id: 1, name: 'OP1' }] }

beforeEach(() => {
  vi.mocked(api.getAdminAnimeThemes).mockResolvedValue(mockThemes)
  vi.mocked(api.getAdminThemeTypes).mockResolvedValue(mockTypes)
  vi.mocked(api.createAdminAnimeTheme).mockResolvedValue({ data: { id: 99, anime_id: 5, theme_type_id: 1, theme_type_name: 'OP1', title: null, created_at: '' } })
  vi.mocked(api.deleteAdminAnimeTheme).mockResolvedValue(undefined)
})

describe('useAdminAnimeThemes', () => {
  it('loads themes and themeTypes on mount', async () => {
    const { result } = renderHook(() =>
      useAdminAnimeThemes({ animeID: 5, authToken: 'token' })
    )
    await waitFor(() => expect(result.current.isLoading).toBe(false))
    expect(api.getAdminAnimeThemes).toHaveBeenCalledWith(5, 'token')
    expect(api.getAdminThemeTypes).toHaveBeenCalledWith('token')
    expect(result.current.themeTypes).toEqual([{ id: 1, name: 'OP1' }])
  })

  it('createTheme calls API with correct animeID and payload', async () => {
    const { result } = renderHook(() =>
      useAdminAnimeThemes({ animeID: 5, authToken: 'token' })
    )
    await waitFor(() => expect(result.current.isLoading).toBe(false))
    act(() => result.current.setNewTypeID(1))
    act(() => result.current.setNewTitle('Again'))
    await act(() => result.current.createTheme())
    expect(api.createAdminAnimeTheme).toHaveBeenCalledWith(5, { theme_type_id: 1, title: 'Again' }, 'token')
  })

  it('deleteTheme calls API with correct IDs', async () => {
    const theme = { id: 10, anime_id: 5, theme_type_id: 1, theme_type_name: 'OP1', title: null, created_at: '' }
    vi.mocked(api.getAdminAnimeThemes).mockResolvedValue({ data: [theme] })
    const { result } = renderHook(() =>
      useAdminAnimeThemes({ animeID: 5, authToken: 'token' })
    )
    await waitFor(() => expect(result.current.themes).toHaveLength(1))
    await act(() => result.current.deleteTheme(10))
    expect(api.deleteAdminAnimeTheme).toHaveBeenCalledWith(5, 10, 'token')
  })
})
```
