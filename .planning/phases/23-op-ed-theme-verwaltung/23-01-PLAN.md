```go
package repository

import (
    "context"
    "errors"

    "github.com/jackc/pgx/v5/pgconn"
    "team4s.v3/backend/internal/models"
)

// ListThemeTypes gibt alle theme_types sortiert nach ID zurück.
func (r *AdminContentRepository) ListThemeTypes(ctx context.Context) ([]models.AdminThemeType, error) {
    rows, err := r.db.Query(ctx, `SELECT id, name FROM theme_types ORDER BY id`)
    if err != nil {
        return nil, err
    }
    defer rows.Close()
    var result []models.AdminThemeType
    for rows.Next() {
        var tt models.AdminThemeType
        if err := rows.Scan(&tt.ID, &tt.Name); err != nil {
            return nil, err
        }
        result = append(result, tt)
    }
    return result, rows.Err()
}

// ListAdminAnimeThemes gibt alle Themes eines Anime zurück, joiniert mit dem Theme-Typ-Namen.
func (r *AdminContentRepository) ListAdminAnimeThemes(ctx context.Context, animeID int64) ([]models.AdminAnimeTheme, error) {
    if animeID <= 0 {
        return nil, ErrNotFound
    }
    exists, err := r.animeExists(ctx, animeID)
    if err != nil {
        return nil, err
    }
    if !exists {
        return nil, ErrNotFound
    }
    rows, err := r.db.Query(ctx, `
        SELECT t.id, t.anime_id, t.theme_type_id, tt.name AS theme_type_name, t.title, t.created_at
        FROM themes t
        JOIN theme_types tt ON tt.id = t.theme_type_id
        WHERE t.anime_id = $1
        ORDER BY t.theme_type_id, t.id
    `, animeID)
    if err != nil {
        return nil, err
    }
    defer rows.Close()
    var result []models.AdminAnimeTheme
    for rows.Next() {
        var th models.AdminAnimeTheme
        if err := rows.Scan(&th.ID, &th.AnimeID, &th.ThemeTypeID, &th.ThemeTypeName, &th.Title, &th.CreatedAt); err != nil {
            return nil, err
        }
        result = append(result, th)
    }
    return result, rows.Err()
}

// CreateAdminAnimeTheme legt ein neues Theme für einen Anime an.
// Gibt ErrNotFound zurück wenn der Anime nicht existiert,
// ErrConflict wenn theme_type_id ungültig ist (FK-Verletzung).
func (r *AdminContentRepository) CreateAdminAnimeTheme(ctx context.Context, animeID int64, input models.AdminAnimeThemeCreateInput) (*models.AdminAnimeTheme, error) {
    if animeID <= 0 {
        return nil, ErrNotFound
    }
    exists, err := r.animeExists(ctx, animeID)
    if err != nil {
        return nil, err
    }
    if !exists {
        return nil, ErrNotFound
    }
    var th models.AdminAnimeTheme
    err = r.db.QueryRow(ctx, `
        INSERT INTO themes (anime_id, theme_type_id, title)
        VALUES ($1, $2, $3)
        RETURNING id, anime_id, theme_type_id, title, created_at
    `, animeID, input.ThemeTypeID, input.Title).Scan(
        &th.ID, &th.AnimeID, &th.ThemeTypeID, &th.Title, &th.CreatedAt,
    )
    if err != nil {
        var pgErr *pgconn.PgError
        if errors.As(err, &pgErr) && pgErr.Code == "23503" {
            return nil, ErrConflict
        }
        return nil, err
    }
    // Resolve theme_type_name for the response
    if err := r.db.QueryRow(ctx, `SELECT name FROM theme_types WHERE id = $1`, input.ThemeTypeID).Scan(&th.ThemeTypeName); err != nil {
        return nil, err
    }
    return &th, nil
}

// UpdateAdminAnimeTheme aktualisiert ein Theme; es werden nur gesetzte (non-nil) Felder überschrieben.
func (r *AdminContentRepository) UpdateAdminAnimeTheme(ctx context.Context, themeID int64, input models.AdminAnimeThemePatchInput) error {
    if themeID <= 0 {
        return ErrNotFound
    }
    // Build partial update
    setClauses := []string{}
    args := []interface{}{}
    argIdx := 1
    if input.ThemeTypeID != nil {
        setClauses = append(setClauses, fmt.Sprintf("theme_type_id = $%d", argIdx))
        args = append(args, *input.ThemeTypeID)
        argIdx++
    }
    if input.Title != nil {
        setClauses = append(setClauses, fmt.Sprintf("title = $%d", argIdx))
        args = append(args, *input.Title)
        argIdx++
    }
    if len(setClauses) == 0 {
        // Nothing to update; verify existence only
        var id int64
        err := r.db.QueryRow(ctx, `SELECT id FROM themes WHERE id = $1`, themeID).Scan(&id)
        if errors.Is(err, pgx.ErrNoRows) {
            return ErrNotFound
        }
        return err
    }
    args = append(args, themeID)
    query := fmt.Sprintf("UPDATE themes SET %s WHERE id = $%d", strings.Join(setClauses, ", "), argIdx)
    tag, err := r.db.Exec(ctx, query, args...)
    if err != nil {
        var pgErr *pgconn.PgError
        if errors.As(err, &pgErr) && pgErr.Code == "23503" {
            return ErrConflict
        }
        return err
    }
    if tag.RowsAffected() == 0 {
        return ErrNotFound
    }
    return nil
}

// DeleteAdminAnimeTheme löscht ein Theme anhand seiner ID.
func (r *AdminContentRepository) DeleteAdminAnimeTheme(ctx context.Context, themeID int64) error {
    if themeID <= 0 {
        return ErrNotFound
    }
    tag, err := r.db.Exec(ctx, `DELETE FROM themes WHERE id = $1`, themeID)
    if err != nil {
        return err
    }
    if tag.RowsAffected() == 0 {
        return ErrNotFound
    }
    return nil
}
```
