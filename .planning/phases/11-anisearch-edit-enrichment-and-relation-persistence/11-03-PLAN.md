export async function enrichAdminAnimeFromAniSearch(
  animeID: number,
  payload: AdminAniSearchEnrichmentRequest,
  authToken?: string,
): Promise<AdminAniSearchEnrichmentResponse> {
  const response = await fetch(
    `${getApiBaseUrl()}/api/v1/admin/anime/${animeID}/enrichment/anisearch`,
    {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        ...withAuthHeaders(authToken),
      },
      body: JSON.stringify(payload),
    },
  )

  if (!response.ok) {
    throw new Error(await parseApiError(response, `API request failed: ${response.status}`))
  }

  return response.json() as Promise<AdminAniSearchEnrichmentResponse>
}

```typescript
async function handleAniSearchLoad() {
  if (!animeID || !aniSearchID.trim()) return
  setAniSearchLoading(true)
  setErrorMessage(null)
  try {
    const response = await enrichAdminAnimeFromAniSearch(
      animeID,
      { anisearch_id: aniSearchID.trim(), locked_fields: aniSearchLocked },
      authToken,
    )
    setAniSearchResult(response.data)
    setSuccessMessage(
      `AniSearch-Enrichment angewendet. ${response.data.relations.length} Relation(en) gespeichert, ${response.data.skipped_relations} uebersprungen.`,
    )
    // Apply draft fields to the existing edit workspace
    // The AnimeEditWorkspace exposes fields via the anime prop which is updated via onSaved.
    // Override mode: set each draft field directly unless it's in aniSearchLocked.
    // We update the anime state so AnimeEditWorkspace re-initializes from it (per D-01 override mode).
    // Per D-03: enrichment updates the edit draft â€” admin must still explicitly save.
    setAnime((prev) => {
      if (!prev) return prev
      const draft = response.data.draft
      return {
        ...prev,
        ...(aniSearchLocked.includes('title') ? {} : { title: draft.title }),
        ...(aniSearchLocked.includes('title_de') ? {} : draft.title_de !== undefined ? { title_de: draft.title_de } : {}),
        ...(aniSearchLocked.includes('title_en') ? {} : draft.title_en !== undefined ? { title_en: draft.title_en } : {}),
        ...(aniSearchLocked.includes('year') ? {} : draft.year !== undefined ? { year: draft.year } : {}),
        ...(aniSearchLocked.includes('description') ? {} : draft.description !== undefined ? { description: draft.description } : {}),
        ...(aniSearchLocked.includes('type') ? {} : draft.type ? { type: draft.type as AnimeDetail['type'] } : {}),
      }
    })
  } catch (error) {
    const message = error instanceof Error && error.message.trim()
      ? error.message
      : 'AniSearch-Enrichment fehlgeschlagen.'
    setErrorMessage(message)
  } finally {
    setAniSearchLoading(false)
  }
}
```
