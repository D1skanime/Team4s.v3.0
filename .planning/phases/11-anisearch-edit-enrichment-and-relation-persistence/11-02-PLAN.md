**In admin_content_anime_create_v2.go, add a private helper:**

```go
// insertAnimeRelationsInTx inserts relations for a newly created anime within an existing transaction.
// Uses ON CONFLICT DO NOTHING to skip duplicates silently.
// Unresolvable relation_type_ids (bad label) are skipped — they were validated upstream.
func insertAnimeRelationsInTx(
    ctx context.Context,
    tx pgx.Tx,
    animeID int64,
    relations []models.AdminAnimeRelationInput,
) error {
    if len(relations) == 0 {
        return nil
    }

    // V1 label → relation_type name mapping
    labelToTypeName := map[string]string{
        "Fortsetzung":     "sequel",
        "Hauptgeschichte": "full-story",
        "Nebengeschichte": "side-story",
        "Zusammenfassung": "summary",
    }

    for _, rel := range relations {
        typeName, ok := labelToTypeName[rel.RelationLabel]
        if !ok {
            // validated upstream, but skip defensively
            continue
        }

        if _, err := tx.Exec(ctx, `
            INSERT INTO anime_relations (source_anime_id, target_anime_id, relation_type_id)
            SELECT $1, $2, rt.id
            FROM relation_types rt
            WHERE rt.name = $3
            ON CONFLICT DO NOTHING
        `, animeID, rel.TargetAnimeID, typeName); err != nil {
            return fmt.Errorf("insert anime relation anime=%d target=%d: %w", animeID, rel.TargetAnimeID, err)
        }
    }

    return nil
}
```

| T-10-07 | Tampering | relations array with arbitrary target_anime_id | mitigate | The FOREIGN KEY constraint on anime_relations.target_anime_id will reject non-existent anime IDs at the DB level; no pre-check needed beyond validation |

</threat_model>
